              if(input == '') {

              return parts[2]+'/'+parts[1]+'/'+parts[0];

                                            olly MizAkita • a year ago

                                            what worked for me was this:

                                            in the main js file (under formatTime:'H:i', for example) i added

                                            formatTimeScroller:'g:i a', /*12 hour fomat without leading zeros + lowercase am/pm [i.e 9:15 am , 17:35 pm etc]*/

                                            and change (further down)

                                            time+= '<div class="xdsoft_time '+classes.join(' ')+'" data-hour="'+h+'" data-minute="'+m+'">'+now.dateFormat(options.formatTime)+'</div>';

                                            to

                                            time+= '<div class="xdsoft_time '+classes.join(' ')+'" data-hour="'+h+'" data-minute="'+m+'">'+now.dateFormat(options.formatTimeScroller)+'</div>';

                                            (you can now also set the format dynamically: something like this:

                                            this.setOptions({
                                            formatTimeScroller: 'g:i A' /*uppercase AM/PM now*/
                                            });

                                            worked for me anyway

                                                                                                                                                    //Get 1 day in milliseconds

                                                                                                                                                    date1 = new Date(date1);

                                                                                                                                                    // Convert both dates to milliseconds

                                                                                                                                                    var date1_ms = date1.getTime();

                                                                                                                                                    var date2_ms = date2.getTime();

                                                                                                                                                            Thanks, Wasim. Your example pointed me in the right direction to address this bug. I'm not sure how this library has changed since your comment, but this was comparing the entered value to datetimepicker's "current_time" which was actually the entered value. The result was that they were actually always equal. When moving back to the current date, the times that have passed were still available when they shouldn't be. Instead, we should compare the selected date to the actual current date (make sure to format the date to match your configuration):

                                                                                                                                                            onSelectDate:function(current_time, $input){

                                                                                                                                                            var currentDate = new Date();
                                                                                                                                                            currentDate = currentDate.dateFormat('d M Y');

                                                                                                                                                            var selectedDate = $input.val(),
                                                                                                                                                            delimiter = ' ',
                                                                                                                                                            start = 0,
                                                                                                                                                            end = 3,
                                                                                                                                                            tokens = selectedDate.split(delimiter).slice(start, end),
                                                                                                                                                            splitDate = tokens.join(delimiter);

                                                                                                                                                            //console.log(currentDate);
                                                                                                                                                            //console.log(splitDate);

                                                                                                                                                            if (splitDate === currentDate)
                                                                                                                                                            {
                                                                                                                                                            //console.log('yes');
                                                                                                                                                            this.setOptions(
                                                                                                                                                            {
                                                                                                                                                            minTime: 0
                                                                                                                                                            });
                                                                                                                                                            }
                                                                                                                                                            else
                                                                                                                                                            {
                                                                                                                                                            //console.log('else');
                                                                                                                                                            this.setOptions(
                                                                                                                                                            {
                                                                                                                                                            minTime: false
                                                                                                                                                            });
                                                                                                                                                            }

                                                                                                                                                            }
